confirm的demo

  • 实例1:

    var confirm = new Confirm({
	tip : '确认要这样做吗',
	onConfirm : function () {
		alert('我确认');
	},
	onCancel : function () {
		alert('我放弃');
	}
});
confirm.on('confirm' , function (e) {
	alert('第二次确认');
}).on('cancel' , function (e) { 
	alert('第二次放弃');
});

  • 实例2:

    • new Confirm({
	tip : '确认要这样做吗',
	onConfirm : function () {
		alert('我确认\n对于alert此回调可缺省');
	},
	type : 'alert'
});

  • 实例3:

    var customHtml = '自定义字符串'；
new Confirm({
	wrap : '.customWrap',
	tip : customHtml,
	onConfirm : function () {
		alert('我确认');
	},
	onCancel : function () {
		alert('放弃');	
	}
});
